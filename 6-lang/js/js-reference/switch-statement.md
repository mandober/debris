# Switch Statement


```js
data = "10";
switch(data) {
  case "1": // multiple cases

  case "1": // multiple cases
  case "2": // resolving to the same statement
  case "3": // may be on separate or the same line
  case "4": case "5":
    console.log("small");
    break;

  // default case need not be last
  default:
    console.log("strange");
  

  case "10": case "11":
    console.log("large");
    break;
}
```
